    ```bash
    git clone https://github.com/Takk8IS/PrometheusWorkana.git
    ```

2. Navigate to the project directory:

    ```bash
    cd PrometheusWorkana
    ```

3. Modify the `prometheusworkana.js` file:

    ```bash
    npm install
    ```
